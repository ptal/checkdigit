00042 { // Check an existing IBM check string s
00043         // which includes a check digit 0 to 9.
00044         // const int n = 8; // n-1 digits, plus 1 nth check digit.

00046         // cout << "Check n = " << n << endl;
00047         // 16 for credit card numbers.
00048         vector<int> v(n);
00049         for (int i = 0; i < n; i++)
00050         { // Convert decimal digit characters in string to integers in vector.
00051                 if (isdigit(s[i]))
00052                 {
00053                         v[i] = s[i] - '0';
00054                 }
00055                 else if (toupper(s[i] == 'X'))
00056                 {
00057                         v[i] = 10;
00058                 }
00059                 else
00060                 { // give up! (assumes no whitespace or punctuation).
00061                         return false;
00062                 }
00063                 //cout << v[i] << ' ';
00064         }
00065         //cout << endl;

00067         for (int i = 0; i < n; i++)

00071                 { // i is even, odd numbered digits (counting 1, 2, 3 ... n) are permuted.

00083         // cout << ' ' << check << endl;
