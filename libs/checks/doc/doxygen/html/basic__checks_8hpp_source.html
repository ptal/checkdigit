00041 template <typename algorithm, typename size_contract, typename iterator>
00042 int compute_checksum(iterator seq_begin, iterator seq_end )
00043 {
00044   unsigned int valid_value_counter = 0;
00045   int checksum = 0 ;
00046   for(; seq_begin != seq_end && !size_contract::reach_one_past_the_end(valid_value_counter) ; ++seq_begin )
00047   {
00048     try{
00049       int current_valid_value = algorithm::translate_to_valid_value( *seq_begin, valid_value_counter );
00050       algorithm::filter_valid_value_with_pos( current_valid_value, valid_value_counter ) ;
00051       algorithm::operate_on_valid_value( current_valid_value, valid_value_counter, checksum ) ;
00052       ++valid_value_counter ;
00053     }
00054     catch( boost::checks::translation_exception ){
00055     }
00056   }
00057   size_contract::respect_size_contract( valid_value_counter );
00058   return checksum ;
00059 }
