00034 bool UPCcheck(std::string s)
00035 { // Check that a UPC includes the correct check digit.
00036   if (s.size() != 12)
00037   { // Must be exactly 11 decimal digits + one decimal check digit.
00038     cout << "length is " << unsigned int(s.size()) << " ! but must be exactly 12."<< endl;
00039     return false;
00040   }
00041   std::vector<int> v (12);
00042   assert(v.size() == 12);
00043   for (int i = 0; i < 12; i++)
00044   {
00045     char d = s[i];
00046     if (isdigit(d))
00047     {
00048       int c = d - int('0');
00049       //v.push_back(c);

00051     }
00052     else
00053     { // Unexpected character in string!
00054       cout << " Unexpected char " << s[i] << endl;
00055       return false;
00056     }
00057   }
00058   int check = (v[0] * 3 + v[1] * 1 + v[2] * 3 + v[3] * 1 + v[4] * 3 + v[5] * 1
00059     + v[6] * 3 + v[7] * 1 + v[8] * 3 + v[9] * 1 + v[10] * 3
00060     + v[11] * 1); // the 12th decimal check digit.
00061   // cout << check << endl;
00062   check %= 10; // modulus 10 because a decimal check digit.
00063   //cout << check << endl;

00067 char UPCcompute(std::string s)
00068 { // Compute UPC check digit.
00069   if (s.size() != 11)
00070   { // Must be exactly 11 decimal digits.

00072     return false;

00081       v[i] = s[i] - int('0');
00082     }
00083     else
00084     { // Unexpected character in string!
00085       cout << " unexpected char " << s[i] << endl;
00086       return false;
00087     }
00088   }

00090     + v[6] * 3 + v[7] * 1 + v[8] * 3 + v[9] * 1 + v[10] * 3);
00091   // cout <<"check int before mod " << check << endl;
