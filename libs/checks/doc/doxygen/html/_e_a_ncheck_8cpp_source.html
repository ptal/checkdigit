00044   if (s.size() != 13)

00055       v[i] = s[i] - '0';

00059       std::cout << " unexpected char " << s[i] << std::endl;

00064     + v[6] * 1 + v[7] * 3 + v[8] * 1 + v[9] * 3 + v[10] * 1 + v[11] * 3
