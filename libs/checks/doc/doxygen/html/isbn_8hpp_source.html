00032 template <class In>
00033 inline bool Is_isbn10(In isbn_begin, In isbn_end)
00034 {
00035   if( isbn_begin == isbn_end ) return false;
00036   In iter = isbn_begin;
00037   int check = 0, i;
00038   for(i=10; i>0 && iter != isbn_end;)
00039   {
00040     if( *iter != '-')
00041     {
00042       if(!isdigit(*iter))
00043         if(i==1 && (*iter == 'x' || *iter == 'X'))  check += 10;
00044         else  return false;
00045       else
00046         check +=  i * (*iter - 48);
00047       --i;
00048     }
00049     ++iter;
00050   }
00051   return (i > 0 || iter != isbn_end) ? false : check % 11 == 0;
00052 }
